```tsx
export function ProjectCreatePage() {
  const navigate = useNavigate();
  const createProject = useCreateProject();

  const handleSubmit = async (data: SimpleProjectFormData) => {
    const project = await createProject.mutateAsync({
      name: data.name,
      description: data.description,
      // tags will be added to API later, store in local state or skip for now
    });
    
    // Navigate to project overview (not edit page)
    navigate(`/projects/${project.project_id}`);
    
    // Show success toast with guidance
    toast.success('Project created! Configure your project using the modules below.');
  };

  return (
    <div className="max-w-2xl mx-auto p-6 space-y-6">
      <header>
        <h1 className="text-2xl font-bold text-foreground">Create Project</h1>
        <p className="text-muted-foreground mt-1">
          Start with the basics. You can configure details after creation.
        </p>
      </header>

      <SimpleProjectForm 
        onSubmit={handleSubmit}
        isSubmitting={createProject.isPending}
      />
    </div>
  );
}
```

**Key changes from current:**
- Remove FolderPlus icon header (simpler)
- Narrower max-width (max-w-2xl vs max-w-5xl)
- Redirect to `/projects/:id` (overview) not `/projects/:id/edit`
- Toast message guides user to configure via modules

Note: Keep the old ProjectForm.tsx for the edit page - it still uses the accordion sections.
